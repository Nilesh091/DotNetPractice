   PROBLEMS OBSERVED IN Sales_Raw TABLE


-- 1) Repeating Groups:
--    Columns ProductNames, Quantities, UnitPrices store
--    multiple comma-separated values → violates First Normal Form (1NF)

-- 2) Customer Data Repetition:
--    CustomerName, Phone, City repeated for every order
--    → causes redundancy and update anomalies

-- 3) SalesPerson Data Mixed with Order Data:
--    SalesPerson depends on Order, not on Product
--    → violates Third Normal Form (3NF)



   NORMALIZED TABLE STRUCTURE


-- Stores unique customer information
CREATE TABLE Customers (
    CustomerID INT IDENTITY PRIMARY KEY,  
    CustomerName VARCHAR(100),
    CustomerPhone VARCHAR(20),
    CustomerCity VARCHAR(50)
);

-- Stores unique sales persons
CREATE TABLE SalesPersons (
    SalesPersonID INT IDENTITY PRIMARY KEY,
    SalesPersonName VARCHAR(100)
);

-- Stores product master data
CREATE TABLE Products (
    ProductID INT IDENTITY PRIMARY KEY,
    ProductName VARCHAR(100),
    UnitPrice INT
);

-- Stores order-level information
CREATE TABLE Orders (
    OrderID INT PRIMARY KEY,
    OrderDate DATE,
    CustomerID INT,        
    SalesPersonID INT,    
    FOREIGN KEY (CustomerID) REFERENCES Customers(CustomerID),
    FOREIGN KEY (SalesPersonID) REFERENCES SalesPersons(SalesPersonID)
);

-- Stores order-product relationship (resolves many-to-many)
CREATE TABLE OrderDetails (
    OrderDetailID INT IDENTITY PRIMARY KEY,
    OrderID INT,           -- References Orders table
    ProductID INT,         -- References Products table
    Quantity INT,
    FOREIGN KEY (OrderID) REFERENCES Orders(OrderID),
    FOREIGN KEY (ProductID) REFERENCES Products(ProductID)
);



   QUESTION 2 – THIRD HIGHEST TOTAL SALES (PER ORDER)


-- Step 1: Calculate total sales per order
WITH OrderTotals AS (
    SELECT 
        OrderID,
        -- Quantity × UnitPrice gives total for each product
        SUM(CAST(q.value AS INT) * CAST(p.value AS INT)) AS TotalSales
    FROM Sales_Raw
    -- Split comma-separated quantities
    CROSS APPLY STRING_SPLIT(Quantities, ',') q
    -- Split comma-separated prices
    CROSS APPLY STRING_SPLIT(UnitPrices, ',') p
    GROUP BY OrderID
),

-- Step 2: Rank orders based on total sales
RankedOrders AS (
    SELECT *,
           DENSE_RANK() OVER (ORDER BY TotalSales DESC) AS rnk
    FROM OrderTotals
)

-- Step 3: Fetch the 3rd highest total sales
SELECT OrderID, TotalSales
FROM RankedOrders
WHERE rnk = 3;



   QUESTION 3 – SALES PERSONS WITH TOTAL SALES > ₹60,000


SELECT 
    SalesPerson,
    -- Calculate total sales per salesperson
    SUM(CAST(q.value AS INT) * CAST(p.value AS INT)) AS TotalSales
FROM Sales_Raw
CROSS APPLY STRING_SPLIT(Quantities, ',') q
CROSS APPLY STRING_SPLIT(UnitPrices, ',') p
GROUP BY SalesPerson
-- Filter only those with sales above 60,000
HAVING SUM(CAST(q.value AS INT) * CAST(p.value AS INT)) > 60000;



   QUESTION 4 – CUSTOMERS SPENDING ABOVE AVERAGE


-- Step 1: Calculate total spending per customer
WITH CustomerTotals AS (
    SELECT 
        CustomerName,
        SUM(CAST(q.value AS INT) * CAST(p.value AS INT)) AS TotalSpent
    FROM Sales_Raw
    CROSS APPLY STRING_SPLIT(Quantities, ',') q
    CROSS APPLY STRING_SPLIT(UnitPrices, ',') p
    GROUP BY CustomerName
)

-- Step 2: Compare customer spending with average spending
SELECT *
FROM CustomerTotals
WHERE TotalSpent > (
    SELECT AVG(TotalSpent) FROM CustomerTotals
);



   QUESTION 5 – STRING & DATE FUNCTIONS


SELECT 
    -- Convert customer name to uppercase
    UPPER(CustomerName) AS CustomerName,
    -- Extract month from order date
    MONTH(CONVERT(DATE, OrderDate)) AS OrderMonth
FROM Sales_Raw
WHERE 
    -- Filter orders from year 2026
    YEAR(CONVERT(DATE, OrderDate)) = 2026
    -- Filter orders from January
    AND MONTH(CONVERT(DATE, OrderDate)) = 1;
